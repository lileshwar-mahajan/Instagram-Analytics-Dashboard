Is Viral = IF('Instagram Analytics'[engagement_rate] > 10, "Viral", "Normal")

dax measure 
	a. Total Interactions: (Sum of all actions)
Total Interactions = SUM('Instagram Analytics'[likes]) + SUM('Instagram Analytics'[comments]) + SUM('Instagram Analytics'[shares]) + SUM('Instagram Analytics'[saves])

	b. Avg. Reach per Post:
Avg Reach = AVERAGE('Instagram Analytics'[reach])

	c. Real Engagement Rate (Impressions based)
Global Engagement Rate = DIVIDE([Total Interactions], SUM('Instagram Analytics'[impressions]), 0)

	d. Saves-to-Like Ratio: (Indicates highly valuable content that users want to keep) 
Saves to Likes Ratio = DIVIDE(SUM('Instagram Analytics'[saves]), SUM('Instagram Analytics'[likes]), 0)

	e. Follower Conversion Rate: (How many people followed after seeing the post)
Follower Conv % = DIVIDE(SUM('Instagram Analytics'[followers_gained]), SUM('Instagram Analytics'[reach]), 0)
